def self.update(local_path,repo,git_path,msg="update file",branch:"",committer:{})
    path = "repos/#{@@owner}/#{repo}/contents/#{git_path}"
    head = {'Authorization':@@authorization}
    body = {
      'content': Base64.encode64(File.binread(local_path)),
      'message': msg,
      'sha': self.get(repo,git_path)['sha'],
      'branch': branch,
      'committer': committer
    }.to_json
    Api_net.req('put', @@domain, path:path, head:head, body:body)
  end
